static size_t app_msg_size(uint16_t payload_len)
{
    return sizeof(app_msg_t) - sizeof(((app_msg_t *)0)->payload) + payload_len;

static int udp_send_raw(uint32_t dst_ip_nbo, uint16_t port, const void *data, size_t len)
{
    int sock = socket(AF_INET, SOCK_DGRAM, IPPROTO_IP);
    if (sock < 0) {
        ESP_LOGE(TAG, "socket failed: errno=%d", errno);
        return -1;
    }
    struct sockaddr_in dest_addr = {0};
    dest_addr.sin_family = AF_INET;
    dest_addr.sin_port = htons(port);
    dest_addr.sin_addr.s_addr = dst_ip_nbo;
    int err = sendto(sock, data, len, 0, (struct sockaddr *)&dest_addr, sizeof(dest_addr));
    close(sock);
    if (err < 0) {
        ESP_LOGE(TAG, "sendto failed: errno=%d", errno);
        return -1;
    }
    return 0;
}

static int udp_send_raw_with_sock(int sock, uint32_t dst_ip_nbo, uint16_t port, const void *data, size_t len)
{
    struct sockaddr_in dest_addr = {0};
    dest_addr.sin_family = AF_INET;
    dest_addr.sin_port = htons(port);
    dest_addr.sin_addr.s_addr = dst_ip_nbo;
    int err = sendto(sock, data, len, 0, (struct sockaddr *)&dest_addr, sizeof(dest_addr));
    if (err < 0) {
        ESP_LOGE(TAG, "sendto failed: errno=%d", errno);
        return -1;
    }
    return 0;
}

static bool get_two_random_nodes(uint32_t *sender_ip,uint32_t *receiver_ip,int *sender_level,int *receiver_level)
{
    uint32_t candidates[32];
    int levels[32];
    uint32_t count = 0;
    for (uint32_t i = 0; i < g_topology_count; i++) {
        if (!g_topology[i].valid) {
            continue;
        }
        if (g_topology[i].ip == 0) {
            continue;
        }
        if (g_topology[i].level == 1) {
            continue;
        }
        candidates[count] = g_topology[i].ip;
        levels[count] = g_topology[i].level;
        count++;
        if (count >= 32) {
            break;
        }
    }
    if (count < 2) {
        return false;
    }
    uint32_t s = esp_random() % count;
    uint32_t r = esp_random() % count;
    while (r == s) {
        r = esp_random() % count;
    }
    *sender_ip = candidates[s];
    *receiver_ip = candidates[r];
    *sender_level = levels[s];
    *receiver_level = levels[r];
    return true;
}

static void ensure_root_in_topology(void)
{
    uint8_t root_mac[6] = {0};
    esp_wifi_get_mac(WIFI_IF_AP, root_mac);
    int idx = find_topology_index_by_ip(inet_addr(ROOT_IP));
    if (idx < 0 && g_topology_count < 32) {
        idx = (int)g_topology_count++;
    }
    if (idx >= 0) {
        g_topology[idx].valid = true;
        g_topology[idx].ip = inet_addr(ROOT_IP);
        memcpy(g_topology[idx].mac, root_mac, 6);
        memset(g_topology[idx].parent_bssid, 0, 6);
        g_topology[idx].level = 1;
        g_topology[idx].parent_rssi = 0;
    }
}

static int find_topology_index_by_ip(uint32_t ip)
{
    for (uint32_t i = 0; i < g_topology_count; i++) {
        if (g_topology[i].valid && g_topology[i].ip == ip) {
            return (int)i;
        }
    }
    return -1;
}

static int find_topology_index_by_mac(const uint8_t mac[6])
{
    for (uint32_t i = 0; i < g_topology_count; i++) {
        if (g_topology[i].valid && memcmp(g_topology[i].mac, mac, 6) == 0) {
            return (int)i;
        }
    }
    return -1;
}

static int build_path_to_node(uint32_t target_ip, uint32_t path_ips[8], uint8_t *path_len)
{
    int idx = find_topology_index_by_ip(target_ip);
    if (idx < 0) {
        return -1;
    }
    int reverse_indices[8];
    int reverse_len = 0;
    int cur = idx;
    while (cur >= 0 && reverse_len < 8) {
        reverse_indices[reverse_len++] = cur;
        if (g_topology[cur].level == 1) {
            break;
        }
        cur = find_topology_index_by_mac(g_topology[cur].parent_bssid);
    }
    if (reverse_len <= 1) {
        *path_len = 0;
        return 0;
    }
    uint8_t out_len = 0;
    for (int i = reverse_len - 2; i >= 0; i--) {
        if (out_len < 8) {
            path_ips[out_len++] = g_topology[reverse_indices[i]].ip;
        }
    }
    *path_len = out_len;
    return 0;
}

static void send_topology_reply_to_root(void)
{
    if (am_i_root()) {
        return;
    }
    uint32_t my_ip = get_my_ip_nbo();
    if (my_ip == 0) {
        return;
    }
    topo_reply_t reply = {0};
    wifi_ap_record_t ap_info = {0};
    uint8_t self_mac[6] = {0};
    esp_wifi_get_mac(WIFI_IF_AP, self_mac);
    reply.node_ip = my_ip;
    memcpy(reply.node_mac, self_mac, 6);
    reply.level = esp_mesh_lite_get_level();
    reply.parent_rssi = -120;
    if (esp_mesh_lite_get_level() > 1) {
        if (esp_wifi_sta_get_ap_info(&ap_info) == ESP_OK) {
            memcpy(reply.parent_bssid, ap_info.bssid, 6);
            reply.parent_rssi = ap_info.rssi;
        }
    }
    app_msg_t reply_msg = {0};
    reply_msg.magic = 0x4D455348;
    reply_msg.type = MSG_TYPE_TOPO_REPLY;
    reply_msg.src_ip = my_ip;
    reply_msg.dst_ip = inet_addr(ROOT_IP);
    reply_msg.payload_len = sizeof(reply);
    memcpy(reply_msg.payload, &reply, sizeof(reply));
    size_t tx_len = app_msg_size(reply_msg.payload_len);
    udp_send_raw(inet_addr(ROOT_IP), UDP_PORT, &reply_msg, tx_len);
}

static void print_reconstructed_path(uint32_t sender_ip, uint32_t receiver_ip)
{
    int sender_idx = find_topology_index_by_ip(sender_ip);
    int receiver_idx = find_topology_index_by_ip(receiver_ip);
    if (sender_idx < 0 || receiver_idx < 0) {
        printf("Path: topology incomplete (missing sender or receiver)\n");
        return;
    }
    int sender_chain[32];
    int receiver_chain[32];
    int sender_len = 0;
    int receiver_len = 0;
    int cur = sender_idx;
    while (cur >= 0 && sender_len < 32) {
        sender_chain[sender_len++] = cur;
        if (g_topology[cur].level == 1) {
            break;
        }
        cur =
        find_topology_index_by_mac(g_topology[cur].parent_bssid);
    }
    cur = receiver_idx;
    while (cur >= 0 && receiver_len < 32) {
        receiver_chain[receiver_len++] = cur;
        if (g_topology[cur].level == 1) {
            break;
        }
        cur =
        find_topology_index_by_mac(g_topology[cur].parent_bssid);
    }
    if (sender_len == 0 || receiver_len == 0) {
        printf("Path: topology incomplete (empty chain)\n");
        return;
    }
    int common_sender = -1;
    int common_receiver = -1;
    for (int i = 0; i < sender_len; i++) {
        for (int j = 0; j < receiver_len; j++) {
            if (sender_chain[i] == receiver_chain[j]) {
                common_sender = i;
                common_receiver = j;
                goto found_common;
            }
        }
    }
    found_common:
    if (common_sender < 0 || common_receiver < 0) {
        printf("Path: topology incomplete (no common ancestor)\n");
        return;
    }
    printf("Path: ");
    for (int i = 0; i <= common_sender; i++) {
        int idx = sender_chain[i];
        struct in_addr ip_addr = { .s_addr = g_topology[idx].ip };
        printf("%s(lvl=%u)", inet_ntoa(ip_addr), g_topology[idx].level);
        if (i < common_sender || common_receiver > 0) {
            printf(" -> ");
        }
    }
    for (int j = common_receiver - 1; j >= 0; j--) {
        int idx = receiver_chain[j];
        struct in_addr ip_addr = { .s_addr = g_topology[idx].ip };
        printf("%s(lvl=%u)", inet_ntoa(ip_addr), g_topology[idx].level);
        if (j > 0) {
            printf(" -> ");
        }
    }
    printf("\n");
}

static void reset_receiver_state(void)

static void send_receiver_progress_summary(bool final_summary)
{
    if (!g_rx_state.active) {
        return;
    }
    receiver_summary_t summary = {0};
    summary.test_id = g_rx_state.test_id;
    summary.sender_ip = g_rx_state.sender_ip;
    summary.receiver_ip = g_rx_state.receiver_ip;
    summary.expected_packets = g_rx_state.expected_packets;
    summary.payload_size = g_rx_state.payload_size;
    summary.received_packets = g_rx_state.received_packets;
    summary.received_bytes = g_rx_state.received_bytes;
    summary.first_rx_us = g_rx_state.first_rx_us;
    summary.last_rx_us = g_rx_state.last_rx_us;

    app_msg_t summary_msg = {0};
    summary_msg.magic = 0x4D455348;
    summary_msg.type = MSG_TYPE_RECEIVER_SUMMARY;
    summary_msg.src_ip = g_rx_state.receiver_ip;
    summary_msg.dst_ip = g_rx_state.root_ip;
    summary_msg.payload_len = sizeof(summary);
    memcpy(summary_msg.payload, &summary, sizeof(summary));
    size_t tx_len = app_msg_size(summary_msg.payload_len);
    int repeat_count = final_summary ? 3 : 1;
    for (int k = 0; k < repeat_count; k++) {
    udp_send_raw(g_rx_state.root_ip, UDP_PORT, &summary_msg,
    tx_len);
    if (repeat_count > 1) {
        vTaskDelay(pdMS_TO_TICKS(20));
    }
}
    if (final_summary) {
        g_rx_state.summary_sent = true;
    }
    g_rx_state.last_progress_report_us = esp_timer_get_time();
}

static void try_send_receiver_summary_on_timeout(void)
{
    if (!g_rx_state.active || g_rx_state.summary_sent) {
        return;
    }
    int64_t now_us = esp_timer_get_time();
    if ((now_us - g_rx_state.last_progress_report_us) >= (RX_PROGRESS_REPORT_MS * 1000)) {
        send_receiver_progress_summary(false);
    }
    int64_t idle_ms = (now_us - g_rx_state.last_rx_us) / 1000;
    uint16_t timeout_ms = g_rx_state.result_timeout_ms;
    if (timeout_ms == 0) {
        timeout_ms = RX_IDLE_TIMEOUT_MS;
    }
    if (idle_ms < timeout_ms) {
        return;
    }
    send_receiver_progress_summary(true);
}

static void print_test_result_if_ready(bool force)
{
    float goodput_rx_kbps = 0.0f;
    if (!am_i_root()) {
        return;
    }
    if (!test_active) {
        return;
    }
    int64_t now_us = esp_timer_get_time();
    int64_t elapsed_ms = (now_us - test_start_us) / 1000;
    bool timeout_reached = (elapsed_ms >= ROOT_RESULT_TIMEOUT_MS);
bool receiver_complete = false;
if (receiver_summary_received &&
    g_best_receiver_summary.expected_packets > 0 &&
    g_best_receiver_summary.received_packets >= g_best_receiver_summary.expected_packets) {
    receiver_complete = true;
}
bool stagnated = false;
if (receiver_summary_received) {
    int64_t no_progress_ms = (now_us - last_receiver_progress_us) / 1000;
    if (no_progress_ms >= ROOT_STAGNATION_MS || same_receiver_count >= ROOT_SAME_COUNT_LIMIT) {
        stagnated = true;
    }
}
bool complete = sender_summary_received && receiver_complete;
bool ready = complete || stagnated || timeout_reached;
if (!force && !ready) {
    return;
}
if (!sender_summary_received && !receiver_summary_received) {
    printf("\n========== TEST RESULT ==========\n");
    printf("Test-ID:                 %" PRIu32 "\n", active_test_id);
    printf("Status:                  NO_SUMMARIES\n");
    printf("Reason:                  Neither sender nor receiver summary reached root\n");
    printf("=================================\n\n");
    test_active = false;
active_test_id = 0;
sender_summary_received = false;
receiver_summary_received = false;
memset(&g_sender_summary, 0, sizeof(g_sender_summary));
memset(&g_best_receiver_summary, 0, sizeof(g_best_receiver_summary));
last_receiver_packets = 0;
same_receiver_count = 0;
last_receiver_progress_us = 0;
return;
}
uint32_t sent_packets = g_sender_summary.sent_packets;
uint32_t received_packets = receiver_summary_received ? g_best_receiver_summary.received_packets : 0;
uint32_t sent_bytes = g_sender_summary.sent_bytes;
uint32_t received_bytes = receiver_summary_received ? g_best_receiver_summary.received_bytes : 0;
uint32_t lost_packets = 0;
float pdr = 0.0f;
float sender_duration_ms = 0.0f;
if (g_sender_summary.send_end_us > g_sender_summary.send_start_us) {
    sender_duration_ms =
        (float)(g_sender_summary.send_end_us - g_sender_summary.send_start_us) / 1000.0f;
}
float receiver_duration_ms = 0.0f;
if (receiver_summary_received &&
    g_best_receiver_summary.last_rx_us > g_best_receiver_summary.first_rx_us) {
    receiver_duration_ms =
        (float)(g_best_receiver_summary.last_rx_us - g_best_receiver_summary.first_rx_us) / 1000.0f;
}
float goodput_kbps = 0.0f;
float root_duration_raw_ms = (float)elapsed_ms;
float root_duration_effective_ms = root_duration_raw_ms;
float goodput_root_kbps = 0.0f;
if (last_receiver_real_progress_us > test_start_us) {
    root_duration_effective_ms =
        (float)(last_receiver_real_progress_us - test_start_us) / 1000.0f;
}
if (receiver_summary_received) {
    if (sent_packets >= received_packets) {
        lost_packets = sent_packets - received_packets;
    }
    if (sent_packets > 0) {
        pdr = ((float)received_packets / (float)sent_packets) * 100.0f;
    }
    if (receiver_duration_ms > 0.0f) {
        goodput_rx_kbps = ((float)received_bytes * 8.0f) / (receiver_duration_ms / 1000.0f) / 1000.0f;
    }
if (receiver_summary_received && root_duration_effective_ms > 0.0f) {
    goodput_root_kbps =
        ((float)received_bytes * 8.0f) /
        (root_duration_effective_ms / 1000.0f) / 1000.0f;
}
}
uint16_t burst_packets = g_sender_summary.packet_count
    ? g_sender_summary.packet_count
    : (receiver_summary_received ? g_best_receiver_summary.expected_packets : 0);
uint16_t payload_size = g_sender_summary.payload_size
    ? g_sender_summary.payload_size
    : (receiver_summary_received ? g_best_receiver_summary.payload_size : 0);
uint32_t sender_ip_nbo = g_sender_summary.sender_ip
    ? g_sender_summary.sender_ip
    : (receiver_summary_received ? g_best_receiver_summary.sender_ip : 0);
uint32_t receiver_ip_nbo = g_sender_summary.receiver_ip
    ? g_sender_summary.receiver_ip
    : (receiver_summary_received ? g_best_receiver_summary.receiver_ip : 0);
struct in_addr s_addr, r_addr;
char sender_ip_str[16];
char receiver_ip_str[16];
s_addr.s_addr = sender_ip_nbo;
r_addr.s_addr = receiver_ip_nbo;
snprintf(sender_ip_str, sizeof(sender_ip_str), "%s", inet_ntoa(s_addr));
snprintf(receiver_ip_str, sizeof(receiver_ip_str), "%s", inet_ntoa(r_addr));
printf("\n========== TEST RESULT ==========\n");
printf("Test-ID:                 %" PRIu32 "\n", active_test_id);
printf("Sender:                  %s\n", sender_ip_str);
printf("Receiver:                %s\n", receiver_ip_str);
printf("Burst packets:           %u\n", burst_packets);
printf("Payload per packet:      %u bytes\n", payload_size);
printf("Sent packets:            %" PRIu32 "\n", sent_packets);
if (receiver_summary_received) {
    printf("Received packets:        %" PRIu32 "\n", received_packets);
    printf("Lost packets:            %" PRIu32 "\n", lost_packets);
    printf("PDR:                     %.2f %%\n", pdr);
    printf("Sent bytes:              %" PRIu32 "\n", sent_bytes);
    printf("Received bytes:          %" PRIu32 "\n", received_bytes);
    printf("Sender duration:         %.2f ms\n", sender_duration_ms);
    printf("Receiver duration:       %.2f ms\n", receiver_duration_ms);
    printf("Goodput (receiver-based): %.2f kbps\n", goodput_rx_kbps);
printf("Root duration raw:         %.2f ms\n", root_duration_raw_ms);
printf("Root duration effective:   %.2f ms\n", root_duration_effective_ms);
printf("Goodput (root-window):     %.2f kbps\n", goodput_root_kbps);

} else {
    printf("Received packets:        unknown (no receiver summary)\n");
    printf("Lost packets:            unknown\n");
    printf("PDR:                     unknown\n");
    printf("Sent bytes:              %" PRIu32 "\n", sent_bytes);
    printf("Received bytes:          unknown (no receiver summary)\n");
    printf("Sender duration:         %.2f ms\n", sender_duration_ms);
    printf("Receiver duration:       unknown\n");
    printf("Goodput:                 unknown\n");
}
const char *status_str = "TIMEOUT";
if (complete) {
    status_str = "COMPLETE";
} else if (!receiver_summary_received && sender_summary_received && timeout_reached) {
    status_str = "NO_RECEIVER_SUMMARY";
} else if (stagnated && sender_summary_received) {
    status_str = "STAGNATED";
} else if (stagnated && !sender_summary_received) {
    status_str = "STAGNATED_NO_SENDER_SUMMARY";
}
printf("Status:                  %s\n", status_str);
print_reconstructed_path(sender_ip_nbo, receiver_ip_nbo);
printf("=================================\n\n");
test_active = false;
active_test_id = 0;
sender_summary_received = false;
receiver_summary_received = false;
memset(&g_sender_summary, 0, sizeof(g_sender_summary));
memset(&g_best_receiver_summary, 0, sizeof(g_best_receiver_summary));
last_receiver_packets = 0;
same_receiver_count = 0;
last_receiver_progress_us = 0;
last_receiver_real_progress_us = 0;
}


static void udp_rx_task(void *arg)
{
    int sock = socket(AF_INET, SOCK_DGRAM, IPPROTO_IP);
    if (sock < 0) {
        ESP_LOGE(TAG, "RX socket failed: errno=%d", errno);
        vTaskDelete(NULL);
        return;
    }
    struct timeval tv = {
        .tv_sec = 0,
        .tv_usec = 200000
    };
    setsockopt(sock, SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv));
    struct sockaddr_in listen_addr = {0};
    listen_addr.sin_family = AF_INET;
    listen_addr.sin_port = htons(UDP_PORT);
    listen_addr.sin_addr.s_addr = htonl(INADDR_ANY);
    if (bind(sock, (struct sockaddr *)&listen_addr,
    sizeof(listen_addr)) < 0) {
        ESP_LOGE(TAG, "bind failed: errno=%d", errno);
        close(sock);
        vTaskDelete(NULL);
        return;
    }
    ESP_LOGI(TAG, "UDP RX listening on port %d", UDP_PORT);
    while (1) {
        print_test_result_if_ready(false);
        try_send_receiver_summary_on_timeout();
        struct sockaddr_in source_addr = {0};
        socklen_t socklen = sizeof(source_addr);
        app_msg_t msg = {0};
        int len = recvfrom(sock, &msg, sizeof(msg), 0, (struct sockaddr *)&source_addr, &socklen);
        if (len < 0) {
            if (errno == EAGAIN || errno == EWOULDBLOCK) {
                continue;
            }
            ESP_LOGE(TAG, "recvfrom failed: errno=%d", errno);
            continue;
        }
        if (msg.magic != 0x4D455348) {
            continue;
        }
	uint32_t my_ip = get_my_ip_nbo();
        if (msg.type == MSG_TYPE_TRIGGER) {
    if (my_ip != msg.dst_ip) {
        continue;
    }
            trigger_msg_t trig = {0};
            if (msg.payload_len < sizeof(trig)) {
                printf("[WARN] TRIGGER payload too small: %u\n", msg.payload_len);
                continue;
            }
            memcpy(&trig, msg.payload, sizeof(trig));
            uint32_t root_ip_from_trigger = msg.src_ip;
	    if (trig.path_len > 0 && trig.current_hop < (trig.path_len - 1)) {
    trig.current_hop++;
    app_msg_t fwd_msg = {0};
    fwd_msg.magic = 0x4D455348;
    fwd_msg.type = MSG_TYPE_TRIGGER;
    fwd_msg.src_ip = msg.src_ip;
    fwd_msg.dst_ip = trig.path_ips[trig.current_hop];
    fwd_msg.payload_len = sizeof(trig);
    memcpy(fwd_msg.payload, &trig, sizeof(trig));
    size_t fwd_len = app_msg_size(fwd_msg.payload_len);
    udp_send_raw(fwd_msg.dst_ip, UDP_PORT, &fwd_msg, fwd_len);
    continue;
}
            sender_summary_t summary = {0};
            summary.test_id = trig.test_id;
            summary.sender_ip = trig.sender_ip;
            summary.receiver_ip = trig.receiver_ip;
            summary.packet_count = trig.packet_count;
            summary.payload_size = trig.payload_size;
            summary.send_start_us = esp_timer_get_time();
int burst_sock = socket(AF_INET, SOCK_DGRAM, IPPROTO_IP);
if (burst_sock < 0) {
    ESP_LOGE(TAG, "burst socket failed: errno=%d", errno);
    continue;
}
            for (uint16_t i = 0; i < trig.packet_count; i++) {
                app_msg_t data_msg = {0};
data_header_t dh = {0};
data_msg.magic = 0x4D455348;
data_msg.type = MSG_TYPE_DATA;
data_msg.seq = i + 1;
data_msg.src_ip = trig.sender_ip;
dh.test_id = trig.test_id;
dh.expected_packets = trig.packet_count;
dh.payload_size = trig.payload_size;
dh.seq = i + 1;
dh.root_ip = root_ip_from_trigger;
dh.result_timeout_ms = trig.result_timeout_ms;
dh.path_len = trig.data_path_len;
memset(dh.path_ips, 0, sizeof(dh.path_ips));
memcpy(dh.path_ips, trig.data_path_ips, sizeof(trig.data_path_ips));
uint32_t data_dst_ip = trig.receiver_ip;
if (trig.data_path_len > 0) {
    data_dst_ip = trig.data_path_ips[0];

    if (data_dst_ip == trig.sender_ip && trig.data_path_len > 1) {
        data_dst_ip = trig.data_path_ips[1];
        dh.current_hop = 1;
    } else {
        dh.current_hop = 0;
    }
} else {
    dh.current_hop = 0;
}
data_msg.dst_ip = data_dst_ip;
size_t app_payload_size = trig.payload_size;
size_t total_payload_size = sizeof(dh) + app_payload_size;
if (total_payload_size > sizeof(data_msg.payload)) {
    app_payload_size = sizeof(data_msg.payload) - sizeof(dh);
    total_payload_size = sizeof(dh) + app_payload_size;
}
data_msg.payload_len = total_payload_size;
memcpy(data_msg.payload, &dh, sizeof(dh));
memset(data_msg.payload + sizeof(dh), 'A' + (i % 26), app_payload_size);
size_t tx_len = app_msg_size(data_msg.payload_len);
if (udp_send_raw_with_sock(burst_sock, data_dst_ip, UDP_PORT, &data_msg, tx_len) == 0) {
    summary.sent_packets++;
    summary.sent_bytes += app_payload_size;
}
                if (trig.send_interval_ms > 0) {
    vTaskDelay(pdMS_TO_TICKS(trig.send_interval_ms));
}
            }
            close(burst_sock);
            summary.send_end_us = esp_timer_get_time();
            app_msg_t summary_msg = {0};
summary_msg.magic = 0x4D455348;
summary_msg.type = MSG_TYPE_SENDER_SUMMARY;
summary_msg.src_ip = trig.sender_ip;
summary_msg.dst_ip = root_ip_from_trigger;
summary_msg.payload_len = sizeof(summary);
memcpy(summary_msg.payload, &summary, sizeof(summary));
size_t tx_len = app_msg_size(summary_msg.payload_len);
for (int k = 0; k < 3; k++) {
    udp_send_raw(root_ip_from_trigger, UDP_PORT, &summary_msg, tx_len);
    vTaskDelay(pdMS_TO_TICKS(20));
}
            continue;
        }
        if (msg.type == MSG_TYPE_DATA) {
            if (my_ip != msg.dst_ip) {
                continue;
            }
            if (msg.payload_len < sizeof(data_header_t)) {
                printf("[WARN] DATA payload too small: %u\n", msg.payload_len);
                continue;
            }
            data_header_t dh = {0};
            memcpy(&dh, msg.payload, sizeof(dh));

            if (dh.path_len > 0 && dh.current_hop < (dh.path_len - 1)) {
    dh.current_hop++;
    app_msg_t fwd_msg = msg;
    fwd_msg.dst_ip = dh.path_ips[dh.current_hop];
    memcpy(fwd_msg.payload, &dh, sizeof(dh));
    size_t fwd_len = app_msg_size(fwd_msg.payload_len);
    udp_send_raw(fwd_msg.dst_ip, UDP_PORT, &fwd_msg, fwd_len);
    continue;
}
            size_t app_payload_size = msg.payload_len - sizeof(dh);
            int64_t now_us = esp_timer_get_time();
            if (!g_rx_state.active || g_rx_state.test_id != dh.test_id) {
                reset_receiver_state();
                g_rx_state.active = true;
                g_rx_state.summary_sent = false;
                g_rx_state.test_id = dh.test_id;
                g_rx_state.sender_ip = msg.src_ip;
                g_rx_state.receiver_ip = my_ip;
                g_rx_state.root_ip = dh.root_ip;
                g_rx_state.expected_packets = dh.expected_packets;
                g_rx_state.payload_size = dh.payload_size;
                g_rx_state.result_timeout_ms = dh.result_timeout_ms;
if (g_rx_state.result_timeout_ms == 0) {
    g_rx_state.result_timeout_ms = RX_IDLE_TIMEOUT_MS;
}
                g_rx_state.received_packets = 0;
                g_rx_state.received_bytes = 0;
                g_rx_state.first_rx_us = now_us;
                g_rx_state.last_rx_us = now_us;
                g_rx_state.last_progress_report_us = now_us;
            }
            if (g_rx_state.received_packets == 0) {
                g_rx_state.first_rx_us = now_us;
            }
            g_rx_state.last_rx_us = now_us;
            g_rx_state.received_packets++;
            g_rx_state.received_bytes += app_payload_size;

            if (!g_rx_state.summary_sent &&
    g_rx_state.expected_packets > 0 &&
    g_rx_state.received_packets >= g_rx_state.expected_packets) {
    send_receiver_progress_summary(true);    
	}
            continue;
        }
        if (msg.type == MSG_TYPE_TOPO_REPLY) {
    if (!am_i_root()) {
        continue;
    }
    topo_reply_t reply = {0};
    if (msg.payload_len < sizeof(reply)) {
        printf("[WARN] TOPO_REPLY payload too small: %u\n", msg.payload_len);
        continue;
    }
    memcpy(&reply, msg.payload, sizeof(reply));
    {
    struct in_addr ip_addr = { .s_addr = reply.node_ip };
    printf("[ROOT-TOPO] got node=%s level=%u mac=" MACSTR " parent=" MACSTR "\n",
           inet_ntoa(ip_addr),
           reply.level,
           MAC2STR(reply.node_mac),
           MAC2STR(reply.parent_bssid));
}
    int idx = find_topology_index_by_ip(reply.node_ip);
    if (idx < 0 && g_topology_count < 32) {
        idx = (int)g_topology_count++;
    }
    if (idx >= 0) {
        g_topology[idx].valid = true;
        g_topology[idx].ip = reply.node_ip;
        memcpy(g_topology[idx].mac, reply.node_mac, 6);
        memcpy(g_topology[idx].parent_bssid, reply.parent_bssid, 6);
        g_topology[idx].level = reply.level;
        g_topology[idx].parent_rssi = reply.parent_rssi;
    }
    continue;
}
        if (msg.type == MSG_TYPE_SENDER_SUMMARY) {
            if (!am_i_root()) {
                continue;
            }
            sender_summary_t summary = {0};
            if (msg.payload_len < sizeof(summary)) {
                printf("[WARN] SENDER_SUMMARY payload too small: %u\n", msg.payload_len);
                continue;
            }
            memcpy(&summary, msg.payload, sizeof(summary));
            if (summary.test_id != active_test_id) {
                continue;
            }
            g_sender_summary = summary;
            sender_summary_received = true;
            printf("[ROOT] sender summary received for test_id=%" PRIu32 "\n", summary.test_id);
            continue;
        }
        if (msg.type == MSG_TYPE_RECEIVER_SUMMARY) {
            if (!am_i_root()) {
                continue;
            }
            receiver_summary_t summary = {0};
            if (msg.payload_len < sizeof(summary)) {
                printf("[WARN] RECEIVER_SUMMARY payload too small: %u\n", msg.payload_len);
                continue;
            }
            memcpy(&summary, msg.payload, sizeof(summary));
            if (summary.test_id != active_test_id) {
                continue;
            }
int64_t now_us = esp_timer_get_time();
if (!receiver_summary_received) {
    receiver_summary_received = true;
    g_best_receiver_summary = summary;
    last_receiver_packets = summary.received_packets;
    last_receiver_progress_us = now_us;
    last_receiver_real_progress_us = now_us;
    same_receiver_count = 0;
} else {
    if (summary.received_packets > g_best_receiver_summary.received_packets) {
        g_best_receiver_summary = summary;
        last_receiver_packets = summary.received_packets;
        last_receiver_progress_us = now_us;
        last_receiver_real_progress_us = now_us;
        same_receiver_count = 0;
    } else if (summary.received_packets == last_receiver_packets) {
        same_receiver_count++;
        last_receiver_progress_us = now_us;
    }
}
printf("[ROOT] receiver summary received for test_id=%" PRIu32
       " packets=%" PRIu32 "/%u same=%" PRIu32 "\n",
       summary.test_id,
       summary.received_packets,
       summary.expected_packets,
       same_receiver_count);
            continue;
        }
    }
}

static esp_err_t esp_storage_init(void)
{
    esp_err_t ret = nvs_flash_init();
    if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
        ESP_ERROR_CHECK(nvs_flash_erase());
        ret = nvs_flash_init();
    }
    return ret;
}

static void wifi_init(void)
{
    wifi_config_t wifi_config;
    memset(&wifi_config, 0x0, sizeof(wifi_config_t));
    esp_bridge_wifi_set_config(WIFI_IF_STA, &wifi_config);
    wifi_config_t wifi_softap_config = { 
.ap = {
.ssid = CONFIG_BRIDGE_SOFTAP_SSID,
.password = CONFIG_BRIDGE_SOFTAP_PASSWORD,
.channel = CONFIG_MESH_CHANNEL,
},
};
esp_bridge_wifi_set_config(WIFI_IF_AP, &wifi_softap_config);
}
void app_wifi_set_softap_info(void)
{
    char softap_ssid[33];
    char softap_psw[64];
    uint8_t softap_mac[6];
    size_t ssid_size = sizeof(softap_ssid);
    size_t psw_size = sizeof(softap_psw);
    esp_wifi_get_mac(WIFI_IF_AP, softap_mac);
    memset(softap_ssid, 0x0, sizeof(softap_ssid));
    memset(softap_psw, 0x0, sizeof(softap_psw));
    if (esp_mesh_lite_get_softap_ssid_from_nvs(softap_ssid, &ssid_size) == ESP_OK) {
        ESP_LOGI(TAG, "Get ssid from nvs: %s", softap_ssid);
    } else {
#ifdef CONFIG_BRIDGE_SOFTAP_SSID_END_WITH_THE_MAC
        snprintf(softap_ssid, sizeof(softap_ssid), "%.25s_%02x%02x%02x", CONFIG_BRIDGE_SOFTAP_SSID, softap_mac[3], softap_mac[4], softap_mac[5]);
#else
        snprintf(softap_ssid, sizeof(softap_ssid), "%.32s", CONFIG_BRIDGE_SOFTAP_SSID);
#endif
        ESP_LOGI(TAG, "Get ssid from nvs failed, set ssid: %s", softap_ssid);
    }
    if (esp_mesh_lite_get_softap_psw_from_nvs(softap_psw, &psw_size) == ESP_OK) {
        ESP_LOGI(TAG, "Get psw from nvs: [HIDDEN]");
    } else {
        strlcpy(softap_psw, CONFIG_BRIDGE_SOFTAP_PASSWORD, sizeof(softap_psw));
        ESP_LOGI(TAG, "Get psw from nvs failed, set psw: [HIDDEN]");
    }
    esp_mesh_lite_set_softap_info(softap_ssid, softap_psw);
}

void app_main()
{
    esp_log_level_set("*", ESP_LOG_INFO);
    ESP_ERROR_CHECK(esp_storage_init());
    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    esp_bridge_create_all_netif();
    wifi_init();
    esp_mesh_lite_config_t mesh_lite_config = ESP_MESH_LITE_DEFAULT_INIT();
    mesh_lite_config.join_mesh_ignore_router_status = true;
#if CONFIG_MESH_ROOT
    mesh_lite_config.join_mesh_without_configured_wifi = false;
#else
    mesh_lite_config.join_mesh_without_configured_wifi = true;
#endif
    esp_mesh_lite_init(&mesh_lite_config);
    app_wifi_set_softap_info();
#if CONFIG_MESH_ROOT
    ESP_LOGI(TAG, "Root node");
    esp_mesh_lite_set_allowed_level(1);
#else
    ESP_LOGI(TAG, "Child node");
    esp_mesh_lite_set_disallowed_level(1);
#endif
    esp_mesh_lite_start();
    TimerHandle_t timer = xTimerCreate("print_system_info", 10000 / portTICK_PERIOD_MS,true, NULL, print_system_info_timercb);
xTimerStart(timer, 0);
xTaskCreate(udp_rx_task, "udp_rx", 8192, NULL, 5, NULL);
#if CONFIG_MESH_ROOT
xTaskCreate(root_trigger_task, "root_trigger", 4096, NULL, 5,NULL);
#endif
}
